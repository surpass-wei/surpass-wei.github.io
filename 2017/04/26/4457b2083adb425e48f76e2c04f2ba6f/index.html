jQuery树组件-jstree使用

发表于 2017-04-26 | 分类于 开发

前言

jstree是一个比较好用的，可定制化支持较好的一个jQuery树组件，最近项目里用到了它，但是发现网上的文档以及使用方式并不适合当前项目的实际开发，又或者是英文看起来比较复杂，因此对这次使用做出相关记录，以备不时之需。

实例代码


1.  `    /**`
2.  `     * 构建课程结构树`
3.  `     */`
4.  `    var jsTreeComponent = {`
5.  `        init: function () {`
6.  `            /*发起ajax请求*/`
7.  `            $.getJSON('/你后台获取树结构的api'), function (data) {`
8.  `                /*这一块的逻辑根据自己的需求进行处理*/`
9.  `                if (data.success != true) {`
10.  `                    $jsTree.html('加载失败');`
11.  `                    return;`
12.  `                } else if (data.data.structureList.length == 0) {`
13.  `                    $jsTree.html('暂无数据');`
14.  `                    return;`
15.  `                }`
16.  `                var structList = data.data.structureList;`
17.18.  `                /*以下是jstree所需对象的特定格式（如果是异步请求，那么是另外一种格式，这里就不多介绍了）*/`
19.  `                var jsTreeBean = {`
20.  `                    id: "string", // 必要的参数`
21.  `                    parent: "string", // 必要的参数`
22.  `                    text: "string", // 节点名称`
23.  `                    icon: "glyphicon glyphicon-tag", // 图标，可使用自定义样式`
24.  `                    state: {`
25.  `                        opened: true,  // 配置是否打开`
26.  `                        disabled: false,  // 配置是否不可选中`
27.  `                        selected: false  // 配置是否选中`
28.  `                    },`
29.  `                    li_attr: {},  // attributes for the generated LI node`
30.  `                    a_attr: {}  // attributes for the generated A node`
31.  `                };`
32.  `                var jsTreeBeanList = [];`
33.34.  `                /*根据后台返回来构建jsTreeBean对象，且填充到jsTreeBeanList数组*/`
35.  `                for (var i = 0; i < structList.length; i++) {`
36.  `                    var current = structList[i];`
37.  `                    var newObj = Object.assign({}, jsTreeBean);`
38.39.  `                    newObj.id = current.id;`
40.  `                    newObj.text = current.name;`
41.  `                    newObj.cusLevel = current.level;    //自定义的层级，0是章，1是节`
42.  `                    if (current.level == 0) {`
43.  `                        newObj.icon = 'glyphicon glyphicon-bookmark';`
44.  `                        newObj.parent = '#';`
45.  `                    } else {`
46.  `                        newObj.parent = current.parentId;`
47.  `                    }`
48.49.  `                    jsTreeBeanList.push(newObj);`
50.  `                }`
51.52.  `                /*核型配置类，用于使用构建好的数组生成树*/`
53.  `                $jsTree.jstree({`
54.  `                    'core': {`
55.  `                        'check_callback': true,`
56.  `                        'data': jsTreeBeanList`
57.  `                    }`
58.  `                });`
59.  `            });`
60.  `        },`
61.62.  `        /*事件对象*/`
63.  `        eventListener: function () {`
64.  `            /*监听'改选'这个事件，其他事件的监听参考官方文档*/`
65.  `            $jsTree.on("changed.jstree", function (e, data) {`
66.  `                /*如果点击的cusLevel是1（节），则获取学习资源*/`
67.  `                if (data.node.original.cusLevel == 1) {`
68.  `                    service.loadLearnResources(data.node.id)`
69.  `                }`
70.  `            });`
71.  `        }`
72.  `    };`
