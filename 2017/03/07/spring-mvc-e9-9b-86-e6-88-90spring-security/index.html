spring mvc集成spring security

发表于 2017-03-07 | 分类于 开发

代码结构

总共涉及到7个类文件和添加角色配置文件以及需要在web.xml中添加一个过滤器（如果是spring boot的话则不需要配置web.xml）

涉及文件及其作用简单分析

  1. role-config.properties：用来配置所有角色及所持权限信息
  2. Role.java:将配置文件中的权限信息分装到java类中，方便处理
  3. RoleHandler.java:构建Role对象列表，以便可以通过配置文件的key获取到相关Role对象
  4. ‘UserDtailsBean.java’:根绝具体业务需求来给用户对象添加（自定义）security需要的属性
  5. ‘UserDetailsServiceCustom’:自定义用户信息获取服务，主要是重写loadUserByUsername方法
  6. AuthenticationProviderCustom.java:自定义授权认证类，当用户登录，则会调用该方法，授权操作就是在这里进行的
  7. CustomSuccessHandler:当登录成功后的处理操作，可以省去或直接跳转到某个页面
  8. SecurityConfig:Security的核心配置类，在这里使用了自定义的AuthenticationProviderCustom和AuthenticationProviderCustom进行用户信息的构建以及授权类,具体配置内容可参考官网（当然，也可以用XML进行配置，而不是Java Config的形式）
  9. web.xml:配置security过滤器（使用spring boot的话无须配置）

具体代码案例

1.在配置文件中配置好各个角色的信息，security会默认解析以ROLE_开头的权限内容作为角色，例如ROLE_ADMIN，则将作为ADMIN角色

3.提供角色权限初始化类


1.  `import com.x.x.common.UserType;`
2.  `import org.springframework.beans.factory.annotation.Value;`
3.  `import org.springframework.stereotype.Component;`
4.5.  `import java.util.ArrayList;`
6.  `import java.util.List;`
7.8.  `/**`
9.  ` * 角色权限管理以及初始化`
10.  ` * 目前是读取配置文件来初始化用户权限`
11.  ` * <p>`
12.  ` * Created by surpass.wei@gmail.com on 2017/3/6.`
13.  ` */`
14.  `@Component`
15.  `public class RoleHandler {`
16.17.  `    @Value("${role.admin}")`
18.  `    private String adminRoleStr;`
19.20.  `    @Value("${role.teacher}")`
21.  `    private String teacherRoleStr;`
22.23.  `    @Value("${role.trainee}")`
24.  `    private String traineeRoleStr;`
25.26.  `    private final List<Role> roleList = new ArrayList<>();`
27.28.  `    /**`
29.  `     * 根据config.properties配置文件中的角色权限控制来初始化roles`
30.  `     */`
31.  `    private void init() {`
32.  `        UserType[] userTypes = UserType.values();`
33.34.  `        Role role;`
35.  `        List<String> privilegeList;`
36.  `        /**`
37.  `         * 循环角色类型获取相应的配置字符串，根据“,”分隔`
38.  `         * 构建Role的List`
39.  `         */`
40.  `        for (UserType userType : userTypes) {`
41.42.  `            String currentRoleStr;  //  当前角色的权限信息`
43.  `            if (userType.equals(UserType.ADMIN)) {`
44.  `                currentRoleStr = adminRoleStr;`
45.  `            } else if (userType.equals(UserType.TEACHER)) {`
46.  `                currentRoleStr = teacherRoleStr;`
47.  `            } else if (userType.equals(UserType.TRAINEE)) {`
48.  `                currentRoleStr = traineeRoleStr;`
49.  `            } else {`
50.  `                continue;`
51.  `            }`
52.  `            if (currentRoleStr.isEmpty()) {`
53.  `                return;`
54.  `            }`
55.56.  `            privilegeList = new ArrayList<>();`
57.  `            String[] roleArr = currentRoleStr.split(",");`
58.  `            for (String s : roleArr) {`
59.  `                privilegeList.add(s.trim());`
60.  `            }`
61.  `            role = new Role(userType.getRoleName(), privilegeList);`
62.63.  `            roleList.add(role);`
64.  `        }`
65.66.  `    }`
67.68.  `    public Role getRole(String roleName) {`
69.  `        if (roleList.isEmpty()) {`
70.  `            this.init();`
71.  `        }`
72.73.  `        if (roleName == null || roleName.isEmpty()) {`
74.  `            return null;`
75.  `        }`
76.77.  `        Role role = new Role();`
78.  `        for (Role temp_role : roleList) {`
79.  `            if (temp_role.getName().compareToIgnoreCase(roleName) == 0) {`
80.  `                role = temp_role;`
81.  `            }`
82.  `        }`
83.84.  `        return role;`
85.  `    }`
86.  `}`

6.自定义的授权认证类


1.  `import com.x.x.entity.User;`
2.  `import com.x.x.service.IUserService;`
3.  `import org.slf4j.Logger;`
4.  `import org.slf4j.LoggerFactory;`
5.  `import org.springframework.security.authentication.AuthenticationProvider;`
6.  `import org.springframework.security.authentication.BadCredentialsException;`
7.  `import org.springframework.security.authentication.UsernamePasswordAuthenticationToken;`
8.  `import org.springframework.security.core.Authentication;`
9.  `import org.springframework.security.core.AuthenticationException;`
10.  `import org.springframework.stereotype.Component;`
11.12.  `import javax.annotation.Resource;`
13.14.  `@Component`
15.  `public class AuthenticationProviderCustom implements AuthenticationProvider {`
16.  `    private Logger logger = LoggerFactory.getLogger(AuthenticationProviderCustom.class);`
17.18.  `    @Resource`
19.  `    private IUserService userService;`
20.21.  `    @Override`
22.  `    public Authentication authenticate(Authentication authentication) throws AuthenticationException {`
23.  `        UsernamePasswordAuthenticationToken token = (UsernamePasswordAuthenticationToken) authentication;`
24.  `        String loginName = token.getName();`
25.  `        String password = token.getCredentials().toString();`
26.  `        User user = userService.getUserByLoginNameAndPwd(loginName, password);`
27.28.  `        UserDetailsBean userDetailsBean;`
29.  `        if(user != null) {`

33.  `            throw new BadCredentialsException("用户名或密码错误");`
34.  `        }`
35.36.  `        return new UsernamePasswordAuthenticationToken(userDetailsBean, password, userDetailsBean.getAuthorities());`
37.  `    }`
38.39.  `    @Override`
40.  `    public boolean supports(Class<?> aClass) {`
